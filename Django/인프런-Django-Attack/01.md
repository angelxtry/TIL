## 3.7.1 버전 설치

pyenv install 3.7.1

## directory 생성

mkdir DjangoAttack
cd DjangoAttack
mkdir first-django

## pyenv setting

pyenv local 3.7.1

## virtualenv이용하여 환경 구성

pyenv virtualenv 3.7.1 django371

python --version

pyenv versions

## django 설치

pip install django

## django project 생성

django-admin startproject firstdjango .

## test run

python manage.py runserver

## Pycharm에서 불러오기

pycharm 실행

## app 생성

python manage.py startapp first
